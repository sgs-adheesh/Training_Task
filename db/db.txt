CREATE SEQUENCE id_seq 
    START WITH 1001;

CREATE TABLE employees (
    id BIGINT PRIMARY KEY DEFAULT nextval('id_seq'),
    name VARCHAR(500) NOT NULL,
    department VARCHAR(500) NOT NULL,
    designation VARCHAR(500) NOT NULL,
    salary DECIMAL NOT NULL
);
